
#   Пример конвертации объектов собственных типов, содержащихся в core:

class K:
    @JSONInitTrojan.json_init  # Эта строчка делает возможным создание объектов этого типа из json
    def __init__(self, a, b):
        self.a = a
        self.b = b

from net_connection.json_serialize import CoreJSONEncoder
from net_connection.json_serialize import CoreJSONDecoder

# объект у отправителя
k_in = K(10, 98)

# json, содержащий все данные
json = CoreJSONEncoder().encode(k_in)

# пересылка json

# получение объекта получателем
k_out = CoreJSONDecoder.decode_json(json)
